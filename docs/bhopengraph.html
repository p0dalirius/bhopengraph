  8class Properties(object):
  9    """
 10    Properties class for storing arbitrary key-value pairs for nodes and edges.
 11    Follows BloodHound OpenGraph schema requirements where properties must be primitive types.
 12    """
 13    
 14    def __init__(self, **kwargs):
 15        """
 16        Initialize Properties with optional key-value pairs.
 17        
 18        Args:
 19          - **kwargs: Key-value pairs to initialize properties
 20        """
 21        self._properties = {}
 22        for key, value in kwargs.items():
 23            self.set_property(key, value)
 24    
 25    def set_property(self, key: str, value):
 26        """
 27        Set a property value. Only primitive types are allowed.
 28        
 29        Args:
 30          - key (str): Property name
 31          - value: Property value (must be primitive type: str, int, float, bool, None, list)
 32        """
 33        if self._is_valid_property_value(value):
 34            self._properties[key] = value
 35        else:
 36            raise ValueError(f"Property value must be a primitive type (str, int, float, bool, None, list), got {type(value)}")
 37    
 38    def get_property(self, key: str, default=None):
 39        """
 40        Get a property value.
 41        
 42        Args:
 43          - key (str): Property name
 44          - default: Default value if key doesn't exist
 45            
 46        Returns:
 47          - Property value or default
 48        """
 49        return self._properties.get(key, default)
 50    
 51    def remove_property(self, key: str):
 52        """
 53        Remove a property.
 54        
 55        Args:
 56          - key (str): Property name to remove
 57        """
 58        if key in self._properties:
 59            del self._properties[key]
 60    
 61    def has_property(self, key: str) -> bool:
 62        """
 63        Check if a property exists.
 64        
 65        Args:
 66          - key (str): Property name to check
 67            
 68        Returns:
 69          - bool: True if property exists, False otherwise
 70        """
 71        return key in self._properties
 72    
 73    def get_all_properties(self) -> dict:
 74        """
 75        Get all properties as a dictionary.
 76        
 77        Returns:
 78          - dict: Copy of all properties
 79        """
 80        return self._properties.copy()
 81    
 82    def clear(self):
 83        """Clear all properties."""
 84        self._properties.clear()
 85    
 86    def _is_valid_property_value(self, value) -> bool:
 87        """
 88        Check if a value is a valid property type.
 89        
 90        Args:
 91          - value: Value to check
 92            
 93        Returns:
 94          - bool: True if value is valid, False otherwise
 95        """
 96        return value is None or isinstance(value, (str, int, float, bool, list))
 97    
 98    def to_dict(self) -> dict:
 99        """
100        Convert properties to dictionary for JSON serialization.
101        
102        Returns:
103          - dict: Properties as dictionary
104        """
105        return self._properties.copy()
106    
107    def __len__(self) -> int:
108        return len(self._properties)
109    
110    def __contains__(self, key: str) -> bool:
111        return key in self._properties
112    
113    def __getitem__(self, key: str):
114        return self._properties[key]
115    
116    def __setitem__(self, key: str, value):
117        self.set_property(key, value)
118    
119    def __delitem__(self, key: str):
120        self.remove_property(key)
121    
122    def __repr__(self) -> str:
123        return f"Properties({self._properties})"

 10class Node(object):
 11    """
 12    Node class representing a node in the OpenGraph.
 13
 14    Follows BloodHound OpenGraph schema requirements with unique IDs, kinds, and properties.
 15
 16    Sources:
 17    - https://bloodhound.specterops.io/opengraph/schema#nodes
 18    - https://bloodhound.specterops.io/opengraph/schema#minimal-working-json
 19    """
 20    
 21    def __init__(self, id: str, kinds: list = None, properties: Properties = None):
 22        """
 23        Initialize a Node.
 24        
 25        Args:
 26          - id (str): Universally unique identifier for the node
 27          - kinds (list): List of node types/classes
 28          - properties (Properties): Node properties
 29        """
 30        if not id:
 31            raise ValueError("Node ID cannot be empty")
 32        
 33        self.id = id
 34        self.kinds = kinds or []
 35        self.properties = properties or Properties()
 36    
 37    def add_kind(self, kind: str):
 38        """
 39        Add a kind/type to the node.
 40        
 41        Args:
 42          - kind (str): Kind/type to add
 43        """
 44        if kind not in self.kinds:
 45            self.kinds.append(kind)
 46    
 47    def remove_kind(self, kind: str):
 48        """
 49        Remove a kind/type from the node.
 50        
 51        Args:
 52          - kind (str): Kind/type to remove
 53        """
 54        if kind in self.kinds:
 55            self.kinds.remove(kind)
 56    
 57    def has_kind(self, kind: str) -> bool:
 58        """
 59        Check if node has a specific kind/type.
 60        
 61        Args:
 62          - kind (str): Kind/type to check
 63            
 64        Returns:
 65          - bool: True if node has the kind, False otherwise
 66        """
 67        return kind in self.kinds
 68    
 69    def set_property(self, key: str, value):
 70        """
 71        Set a property on the node.
 72        
 73        Args:
 74          - key (str): Property name
 75          - value: Property value
 76        """
 77        self.properties[key] = value
 78    
 79    def get_property(self, key: str, default=None):
 80        """
 81        Get a property from the node.
 82        
 83        Args:
 84          - key (str): Property name
 85          - default: Default value if property doesn't exist
 86            
 87        Returns:
 88          - Property value or default
 89        """
 90        return self.properties.get_property(key, default)
 91    
 92    def remove_property(self, key: str):
 93        """
 94        Remove a property from the node.
 95        
 96        Args:
 97          - key (str): Property name to remove
 98        """
 99        self.properties.remove_property(key)
100    
101    def to_dict(self) -> dict:
102        """
103        Convert node to dictionary for JSON serialization.
104        
105        Returns:
106          - dict: Node as dictionary following BloodHound OpenGraph schema
107        """
108        node_dict = {
109            "id": self.id,
110            "kinds": self.kinds.copy(),
111            "properties": self.properties.to_dict()
112        }
113        return node_dict
114    
115    def __eq__(self, other):
116        """Check if two nodes are equal based on their ID."""
117        if isinstance(other, Node):
118            return self.id == other.id
119        return False
120    
121    def __hash__(self):
122        """Hash based on node ID for use in sets and as dictionary keys."""
123        return hash(self.id)
124    
125    def __repr__(self) -> str:
126        return f"Node(id='{self.id}', kinds={self.kinds}, properties={self.properties})"

 11class Edge(object):
 12    """
 13    Edge class representing a directed edge in the OpenGraph.
 14
 15    Follows BloodHound OpenGraph schema requirements with start/end nodes, kind, and properties.
 16    All edges are directed and one-way as per BloodHound requirements.
 17
 18    Sources:
 19    - https://bloodhound.specterops.io/opengraph/schema#edges
 20    - https://bloodhound.specterops.io/opengraph/schema#minimal-working-json
 21    """
 22    
 23    def __init__(self, start_node_id: str, end_node_id: str, kind: str, properties: Properties = None):
 24        """
 25        Initialize an Edge.
 26        
 27        Args:
 28          - start_node_id (str): ID of the source node
 29          - end_node_id (str): ID of the destination node
 30          - kind (str): Type/class of the edge relationship
 31          - properties (Properties): Edge properties
 32        """
 33        if not start_node_id:
 34            raise ValueError("Start node ID cannot be empty")
 35        if not end_node_id:
 36            raise ValueError("End node ID cannot be empty")
 37        if not kind:
 38            raise ValueError("Edge kind cannot be empty")
 39        
 40        self.start_node_id = start_node_id
 41        self.end_node_id = end_node_id
 42        self.kind = kind
 43        self.properties = properties or Properties()
 44    
 45    def set_property(self, key: str, value):
 46        """
 47        Set a property on the edge.
 48        
 49        Args:
 50          - key (str): Property name
 51          - value: Property value
 52        """
 53        self.properties[key] = value
 54    
 55    def get_property(self, key: str, default=None):
 56        """
 57        Get a property from the edge.
 58        
 59        Args:
 60          - key (str): Property name
 61          - default: Default value if property doesn't exist
 62            
 63        Returns:
 64          - Property value or default
 65        """
 66        return self.properties.get_property(key, default)
 67    
 68    def remove_property(self, key: str):
 69        """
 70        Remove a property from the edge.
 71        
 72        Args:
 73          - key (str): Property name to remove
 74        """
 75        self.properties.remove_property(key)
 76    
 77    def to_dict(self) -> dict:
 78        """
 79        Convert edge to dictionary for JSON serialization.
 80        
 81        Returns:
 82          - dict: Edge as dictionary following BloodHound OpenGraph schema
 83        """
 84        edge_dict = {
 85            "kind": self.kind,
 86            "start": {
 87                "value": self.start_node_id,
 88                "match_by": "id"
 89            },
 90            "end": {
 91                "value": self.end_node_id,
 92                "match_by": "id"
 93            }
 94        }
 95        
 96        # Only include properties if they exist and are not empty
 97        if self.properties and len(self.properties) > 0:
 98            edge_dict["properties"] = self.properties.to_dict()
 99        
100        return edge_dict
101    
102    def get_start_node_id(self) -> str:
103        """
104        Get the start node ID.
105        
106        Returns:
107          - str: Start node ID
108        """
109        return self.start_node_id
110    
111    def get_end_node_id(self) -> str:
112        """
113        Get the end node ID.
114        
115        Returns:
116          - str: End node ID
117        """
118        return self.end_node_id
119    
120    def get_kind(self) -> str:
121        """
122        Get the edge kind/type.
123        
124        Returns:
125          - str: Edge kind
126        """
127        return self.kind
128    
129    def __eq__(self, other):
130        """Check if two edges are equal based on their start, end, and kind."""
131        if isinstance(other, Edge):
132            return (self.start_node_id == other.start_node_id and 
133                   self.end_node_id == other.end_node_id and 
134                   self.kind == other.kind)
135        return False
136    
137    def __hash__(self):
138        """Hash based on start, end, and kind for use in sets and as dictionary keys."""
139        return hash((self.start_node_id, self.end_node_id, self.kind))
140    
141    def __repr__(self) -> str:
142        return f"Edge(start='{self.start_node_id}', end='{self.end_node_id}', kind='{self.kind}', properties={self.properties})"

164    def findPaths(self, start_id: str, end_id: str, max_depth: int = 10) -> List[List[str]]:
165        """
166        Find all paths between two nodes using BFS.
167        
168        Args:
169          - start_id (str): Starting node ID
170          - end_id (str): Target node ID
171          - max_depth (int): Maximum path length to search
172            
173        Returns:
174          - List[List[str]]: List of paths, where each path is a list of node IDs
175        """
176        if start_id not in self.nodes or end_id not in self.nodes:
177            return []
178        
179        if start_id == end_id:
180            return [[start_id]]
181        
182        paths = []
183        queue = [(start_id, [start_id])]
184        visited = {start_id}
185        
186        while queue and len(queue[0][1]) <= max_depth:
187            current_id, path = queue.pop(0)
188            
189            for edge in self.getEdgesFromNode(current_id):
190                next_id = edge.end_node_id
191                if next_id not in visited:
192                    new_path = path + [next_id]
193                    if next_id == end_id:
194                        paths.append(new_path)
195                    else:
196                        visited.add(next_id)
197                        queue.append((next_id, new_path))
198        
199        return paths

201    def getConnectedComponents(self) -> List[Set[str]]:
202        """
203        Find all connected components in the graph.
204        
205        Returns:
206          - List[Set[str]]: List of connected component sets
207        """
208        visited = set()
209        components = []
210        
211        for node_id in self.nodes:
212            if node_id not in visited:
213                component = set()
214                stack = [node_id]
215                
216                while stack:
217                    current = stack.pop()
218                    if current not in visited:
219                        visited.add(current)
220                        component.add(current)
221                        
222                        # Add all adjacent nodes
223                        for edge in self.getEdgesFromNode(current):
224                            if edge.end_node_id not in visited:
225                                stack.append(edge.end_node_id)
226                        for edge in self.getEdgesToNode(current):
227                            if edge.start_node_id not in visited:
228                                stack.append(edge.start_node_id)
229                
230                components.append(component)
231        
232        return components

239          - List[str]: List of validation error messages
240        """
241        errors = []
242        
243        # Check for orphaned edges
244        for edge in self.edges:

246                errors.append(f"Edge {edge.kind} references non-existent start node: {edge.start_node_id}")

249

253            if not self.getEdgesFromNode(node_id) and not self.getEdgesToNode(node_id):
254                isolated_nodes.append(node_id)
255        
256        if isolated_nodes:

258        
259        return errors
